BEGIN

  MERGE INTO MPS.event_upload_status trg

  USING (

    WITH fc(id,

    eus_name,

    eus_descrtiptor) AS

     (

      SELECT 1 id,

              'New' eus_name,

              'A New event. Waiting for processing' eus_descrtiptor

        FROM dual

      UNION ALL

      SELECT 2 id,

              'InProgress' eus_name,

              'Event taken to work' eus_descrtiptor

        FROM dual

      UNION ALL

      SELECT 3 id, 'Canceled' eus_name, 'Event was canceled' eus_descrtiptor

        FROM dual

      UNION ALL

      SELECT 4 id,

              'Error' eus_name,

              'Processing completed with an error. Check ErrorCode and ErrorDescription for more details' eus_descrtiptor

        FROM dual

      UNION ALL

      SELECT 5 id,

              'Processed' eus_name,

              'Processing completed successfully' eus_descrtiptor

        FROM dual)

    SELECT *

      FROM fc) src

        ON (trg.id = src.id) WHEN MATCHED THEN

      UPDATE

         SET trg.eus_name        = src.eus_name,

             trg.eus_descrtiptor = src.eus_descrtiptor

    WHEN NOT MATCHED THEN

      INSERT

        (id, eus_name, eus_descrtiptor)

      VALUES

        (src.id, src.eus_name, src.eus_descrtiptor);

  COMMIT;

END;
